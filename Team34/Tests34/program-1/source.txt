procedure proc1 {
	read x;
	read y;
}

procedure proc2 {
	count = 0;
	X = 0;
	Y = 0;
	call proc1;
	while ((x != 0) && (y != 0)) {
		count = count + 1;
		X = X + x;
		Y = Y + y;
		call proc1;
	}
	if (count == 0) then {
		flag = 1;
	} else {
		X = X / count;
		Y = Y / count;
	}
	normSq = X * X + Y * Y;
}
procedure proc3 {
	print flag;
	print X;
	print Y;
	print normSq;
}

procedure main {
	flag = 0;
	call proc2;
	call proc3;
}
